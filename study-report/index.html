How to setup an Express project with Typescript

Install the typescript package.

$ npm install -g typescript
Create package.json file.
$ npm init

About the Typescript node package

Node.js is an engine that runs Javascript and not Typescript. The node Typescript package allows you to transpile your .ts files to .js scripts.

add "tsc": "tsc" in package.json

$ tsc app.ts

# Using number, boolean and string types.

TS
                                            
                                                function cal(num1: number, num2: number, printResult: boolean, showResult: string){
                                                    if(typeof num1 === 'number' && printResult === true){
                                                        return num1 * num2;
                                                    } else {
                                                        throw new Error('Wrong type');
                                                    }
                                                }
                                                const num1 = 2;
                                                const num2 = 30.5;
                                                const printResult = true;
                                                const showResult = 'result:'
                                                const result = cal(num1, num2, printResult, showResult);
                                                console.log(showResult + result);

                                                var Roles;
                                                (function (Roles) {
                                                    Roles["user"] = "Jimmy Depp";
                                                    Roles[Roles["adminNumber"] = 25] = "adminNumber";
                                                })(Roles || (Roles = {}));
                                                ;
                                                var preson = {
                                                    name: 'Johnny Depp',
                                                    hobbies: ['Sleep', 'Fishing', 'Running'],
                                                    role: Roles.adminNumber
                                                };

Function Types & Callbacks

TypeScript supports callback functions to make your program asynchronous, a function which is scheduled to be called after some asynchronous processing is completed. 
                            
function addAndHandle(n1:number, n2: number, cb: (num: number) => void){
    const result = n1 + n2;
    cb(result);
}

addAndHandle(10,20, (result) => {
    console.log(result);
})
